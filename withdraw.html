async function withdraw() { const name = document.getElementById("name").value; const selected = document.getElementById("selectedItem").value; const search = document.getElementById("itemSearch").value; const qty = parseInt(document.getElementById("qty").value); const item = selected || search; // 🔥 fallback if (!name || !item || !qty) { alert("Fill all fields properly"); return; } await fetch(API_URL, { method: "POST", body: JSON.stringify({ name, item, qty, change: -qty, action: "withdrew" }) }); alert("Withdraw successful"); }
