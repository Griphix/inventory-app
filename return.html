async function submitReturn() { const name = document.getElementById("name").value; const item = document.getElementById("itemSelect").value; const qty = document.getElementById("qty").value; if (!name || !item || !qty) { alert("Fill all fields"); return; } const url = `${API_URL}?action=return&name=${encodeURIComponent(name)}&item=${encodeURIComponent(item)}&qty=${qty}`; try { await fetch(url); alert("Returned successfully!"); location.href = "index.html"; } catch (err) { console.error(err); alert("Error submitting return"); } }
